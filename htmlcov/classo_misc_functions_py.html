51def theoretical_lam(n, d): 

52 r"""Theoretical lambda as a function of the dimensions of the problem 

53 

54 This function returns (with :math:`\phi = erf`) : 

55 

56 :math:`4/ \sqrt{n} \phi^{-1}(1 - 2x)` such that :math:`x = 4/d ( \phi^{-1}(1-2x)4 + \phi^{-1}(1-2x)^2 )` 

57 

58 Which is the same (thanks to formula : :math:`norm^{-1}(1-t) = \sqrt{2}\phi^{-1}(1-2t)` ) as : 

59 

60 :math:`\sqrt{2/n} * norm^{-1}(1-k/p)` such that :math:`k = norm^{-1}(1 - k/p)^4 + 2norm^{-1}(1 - k/p)^2` 

61 

62 Args: 

63 n (int) : number of sample 

64 d (int) : number of variables 

65 

66 Returns: 

67 float : theoretical lambda 

68 

69 """ 

70 x = 0.0 

71 dx = 0.1 

72 for i in range(10): 

73 bo = True 

74 while bo: 

75 x += dx 

76 f = erfinv(1 - 2 * x) 

77 xd = 4 / d * (f ** 4 + f ** 2) 

78 bo = xd > x 

79 x = x - dx 

80 dx = dx / 10 

81 return 2 * f / np.sqrt(n)

240def clr(array, coef = 0.5): 

241 """Centered-Log-Ratio transformation 

242 

243 Set all non positive entry to a constant coef. Then compute the log of each component. Then substract the mean of each column. 

244 

245 Args: 

246 array (ndarray) : matrix nxd 

247 coef (float, optional) : Value to replace the zero values 

248 

249 Returns: 

250 ndarray : clr transformed matrix nxd 

251 

252 """ 

253 M = np.copy(array) 

254 null_set = M <= 0.0 

255 M[null_set] = np.ones(M[null_set].shape) * coef 

256 M = np.log(M) 

257 return M - np.mean(M, axis = 0)

266 

267 if intercept: 

268 A1, C1, y = cmatrices 

269 A = np.concatenate([np.ones((len(A1), 1)), A1], axis=1) 

270 C = np.concatenate([np.zeros((len(C1), 1)), C1], axis=1) 

271 

272 else:

274 

275 k = len(C) 

276 d = len(A[0]) 

277 M1 = np.concatenate([A.T.dot(A), C.T], axis=1)

279 M = np.concatenate([M1, M2], axis=0)

281 sol = LA.lstsq( M, b, rcond=None)[0] 

282 beta = sol[: d]

343def rescale(matrices): 

344 ""Function that rescale the matrix and returns its scale 

345 

346 Substract the mean of y, then divides by its norm. Also divide each colomn of X by its norm. 

347 This will change the solution, not only by scaling it, because then the L1 norm will affect every component equally (and not only the variables with big size) 

348 

349 Args: 

350 matrices (tuple) : tuple of three ndarray matrices corresponding to (X,C,y) 

351 

352 Returns: 

353 tuple : tuple of the three corresponding matrices after normalization

355

357 (X, C, y) = matrices 

358 

359 my = sum(y) / len(y) 

360 lX = [LA.norm(X[:, j]) for j in range(len(X[0]))]

372def hub(r, rho): 

373 h = 0 

374 for j in range(len(r)): 

375 if abs(r[j]) < rho: 

376 h += r[j] ** 2 

377 elif r[j] > 0: 

378 h += (2 * r[j] - rho) * rho 

379 else:
